Usage example

Classic usage

var request = new TimeSeriesRequest(new DateTimeOffsetInterval(new DateTimeOffset(new DateTime(2020, 01, 01)), new DateTime(2020, 02, 01), true, true));
request.TimePartitioning = new MonthPartitioning(1);

request.DataSeriesSources.Add(
    new DataSeriesSource(Eui64.ParseExact("FE802127FFFFEEB6", Eui64Presentation.Default), 1, 1));
request.DataSeriesSources.Add(
    new DataSeriesSource(Eui64.ParseExact("FE802127FFFFEEB6", Eui64Presentation.Default), 1, 1));

request.TimeAggregation = new WeightedTimeAggregation(TimeWeighting.TimeToNext, WeightedStatisticFunction.ArithmeticMean);

request.ValueFilters.Add(new MinMaxValueFilter(new DoubleInterval(0, 1000, true, true)));
request.ValueFilters.Add(new MinMaxValueFilter(new DoubleInterval(0, 700, true, true)));

request.ResourceAggregation = ResourceAggregationFunction.ArithmeticMean;
request.TransmissionModeFilter.Add(TransmissionMode.Occasional);
